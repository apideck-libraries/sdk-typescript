# FormField

## Example Usage

```typescript
import { FormField } from "@apideck/unify/models/components";

let value: FormField = {
  id: "channel",
  label: "Channel",
  placeholder: "Select a channel",
  description: "Slack channel to push notification to",
  type: "select",
  required: true,
  customField: false,
  disabled: false,
  hidden: false,
  deprecated: false,
  sensitive: false,
  prefix: "https://",
  suffix: ".shopify.com",
  options: [
    {
      label: "General Channel",
      value: "general",
    },
  ],
};
```
