# DataScopes

## Example Usage

```typescript
import { DataScopes } from "@apideck/unify/models/components";

let value: DataScopes = {
  enabled: true,
  updatedAt: "2025-04-15T14:30:00Z",
  resources: {
    "hris.employees": {
      "id": {
        read: true,
        write: false,
      },
      "first_name": {
        read: true,
        write: true,
      },
      "last_name": {
        read: true,
        write: true,
      },
      "email": {
        read: true,
        write: true,
      },
      "addresses[].street": {
        read: true,
        write: false,
      },
      "addresses[].city": {
        read: true,
        write: false,
      },
      "employment.job_title": {
        read: true,
        write: true,
      },
      "custom_fields[].employee_number": {
        read: true,
        write: false,
      },
    },
    "hris.departments": {
      "id": {
        read: true,
        write: false,
      },
      "name": {
        read: true,
        write: true,
      },
      "code": {
        read: true,
        write: false,
      },
    },
  },
};
```
