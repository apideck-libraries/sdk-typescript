## list

List pipelines

### Example Usage

```typescript
import { Apideck } from "@apideck/unify";

const apideck = new Apideck({
  apiKey: process.env["APIDECK_API_KEY"] ?? "",
  consumerId: "test-consumer",
  appId: "dSBdXd2H6Mqwfg0atXHXYcysLJE9qyn1VwBtXHX",
});

async function run() {
  const result = await apideck.crm.pipelines.list({
    serviceId: "salesforce",
    passThrough: {
      "search": "San Francisco",
    },
    fields: "id,updated_at",
  });

  for await (const page of result) {
    // Handle the page
    console.log(page);
  }
}

run();
```

## create

Create pipeline

### Example Usage

```typescript

});

  const result = await apideck.crm.pipelines.create({
    serviceId: "salesforce",
    pipeline: {
      id: "default",
      name: "Sales Pipeline",
      currency: "USD",
      archived: false,
      active: false,
      displayOrder: 1,
      winProbabilityEnabled: true,
      stages: [
        {
          name: "Contract Sent",
          value: "CONTRACT_SENT",
          winProbability: 50,
          displayOrder: 1,
        },
        {
          name: "Contract Sent",
          value: "CONTRACT_SENT",
          winProbability: 50,
          displayOrder: 1,
        },
        {
          name: "Contract Sent",
          value: "CONTRACT_SENT",
          winProbability: 50,
          displayOrder: 1,
        },
      ],
      passThrough: [
        {
          serviceId: "<id>",
          extendPaths: [
            {
              path: "$.nested.property",
              value: {
                "TaxClassificationRef": {
                  "value": "EUC-99990201-V1-00020000",
                },
              },
            },
          ],
        },

    },

  console.log(result);
}
```
