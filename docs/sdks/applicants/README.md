## list

  apiKey: process.env["APIDECK_API_KEY"] ?? "",
  consumerId: "test-consumer",

});

async function run() {
  const result = await apideck.ats.applicants.list({
    serviceId: "salesforce",
    filter: {
      jobId: "1234",
    },
    passThrough: {
      "search": "San Francisco",
    },
    fields: "id,updated_at",
  });

  for await (const page of result) {
    // Handle the page

  }
}
